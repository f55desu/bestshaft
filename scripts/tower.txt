var a = 100;
var b = 100;
var h = 10;
var n = 20;
var k = 0.8;
   
var x = a;
var y = b;
var matrix = [1, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0];
var base_csys = CONSTRAINTS_Create_Base_CoorSys(matrix); 

for (var i = 1; i <= n; i++)
{
   SOLID_Create_Block("block_" + i, a/2 - x/2, b/2 - y/2, h * i, x, y, h);   
   x *= k;
   y *= k;      
}
